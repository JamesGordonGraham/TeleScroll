<div id="controls">
  <button onclick="toggleScroll()"><span id="toggleText">Pause</span></button>
  <label>Speed:
    <input type="range" min="0.5" max="5" value="1" step="0.1" id="speedControl" onchange="updateSpeed()" />
  </label>
  <button onclick="jumpToMarker(0)">Go to Marker 1</button>
  <button onclick="jumpToMarker(1)">Go to Marker 2</button>
</div>

<script>
  const scrollText = document.getElementById('scrollText');
  const speedSlider = document.getElementById('speedControl');
  const toggleBtn = document.getElementById('toggleText');

  let scrollSpeed = parseFloat(speedSlider.value);
  let scrollPos = 100; // percentage
  let scrolling = true;

  function updateSpeed() {
    scrollSpeed = parseFloat(speedSlider.value);
  }

  function toggleScroll() {
    scrolling = !scrolling;
    toggleBtn.innerText = scrolling ? 'Pause' : 'Resume';
  }

  function jumpToMarker(index) {
    const markers = document.querySelectorAll('.marker');
    if (markers[index]) {
      const markerTop = markers[index].offsetTop;
      scrollPos = 100 * (markerTop / scrollText.scrollHeight);
    }
  }

  function scrollLoop() {
    if (scrolling) {
      scrollPos -= scrollSpeed * 0.1;
      if (scrollPos < -100) scrollPos = 100;
      scrollText.style.top = scrollPos + '%';
    }
    requestAnimationFrame(scrollLoop);
  }

  scrollLoop();
</script>
